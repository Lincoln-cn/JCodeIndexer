**File**: `src/main/java/com/sodlinken/jindexer/parser/JavaParserAdapter.java`

Add a new private method `extractApiRoutes`:

```java
private void extractApiRoutes(ClassOrInterfaceDeclaration cd,
                               String classBasePath,
                               List<ApiRoute> apiRoutes,
                               String filePath) {
    // Check if class is a REST controller
    boolean isRestController = cd.getAnnotationByName("RestController").isPresent()
        || cd.getAnnotationByName("Controller").isPresent();
    if (!isRestController) return;

    // Extract class-level @RequestMapping base path
    String basePath = classBasePath;

    // For each method, check for mapping annotations
    for (MethodDeclaration md : cd.getMethods()) {
        String methodPath = "";
        String httpMethod = null;

        // Check @GetMapping, @PostMapping, etc.
        var getMapping = md.getAnnotationByName("GetMapping");
        var postMapping = md.getAnnotationByName("PostMapping");
        var putMapping = md.getAnnotationByName("PutMapping");
        var deleteMapping = md.getAnnotationByName("DeleteMapping");
        var patchMapping = md.getAnnotationByName("PatchMapping");
        var requestMapping = md.getAnnotationByName("RequestMapping");

        if (getMapping.isPresent()) {
            httpMethod = "GET";
            methodPath = extractMappingPath(getMapping.get());
        } else if (postMapping.isPresent()) {
            httpMethod = "POST";
            methodPath = extractMappingPath(postMapping.get());
        } else if (putMapping.isPresent()) {
            httpMethod = "PUT";
            methodPath = extractMappingPath(putMapping.get());
        } else if (deleteMapping.isPresent()) {
            httpMethod = "DELETE";
            methodPath = extractMappingPath(deleteMapping.get());
        } else if (patchMapping.isPresent()) {
            httpMethod = "PATCH";
            methodPath = extractMappingPath(patchMapping.get());
        } else if (requestMapping.isPresent()) {
            methodPath = extractMappingPath(requestMapping.get());
            httpMethod = extractRequestMappingMethod(requestMapping.get());
        }

        if (httpMethod != null) {
            String fullPath = normalizePath(basePath, methodPath);
            String qualifiedName = cd.getFullyQualifiedName().orElse("") + "." + md.getNameAsString();
            apiRoutes.add(new ApiRoute(0, 0, httpMethod, fullPath, basePath, methodPath,
                                        filePath, md.getBegin().map(p -> p.line).orElse(0)));
        }
    }
}
```

Add helper methods:

```java
private String extractMappingPath(AnnotationExpr ann) {
    if (ann instanceof SingleMemberAnnotationExpr single) {
        return stripQuotes(single.getMemberValue().toString());
    } else if (ann instanceof NormalAnnotationExpr normal) {
        for (var pair : normal.getPairs()) {
            if ("value".equals(pair.getNameAsString()) || "path".equals(pair.getNameAsString())) {
                return stripQuotes(pair.getValue().toString());
            }
        }
    }
    return "";
}

private String extractRequestMappingMethod(AnnotationExpr ann) {
    if (ann instanceof NormalAnnotationExpr normal) {
        for (var pair : normal.getPairs()) {
            if ("method".equals(pair.getNameAsString())) {
                String val = stripQuotes(pair.getValue().toString());
                // Handle RequestMethod.GET -> GET
                if (val.contains(".")) {
                    val = val.substring(val.lastIndexOf('.') + 1);
                }
                return val;
            }
        }
    }
    return "GET"; // Default when @RequestMapping has no method attribute
}

private String stripQuotes(String s) {
    if (s.startsWith("\"") && s.endsWith("\"")) {
        return s.substring(1, s.length() - 1);
    }
    return s;
}

private String normalizePath(String base, String method) {
    String b = base.endsWith("/") ? base.substring(0, base.length() - 1) : base;
    String m = method.startsWith("/") ? method : "/" + method;
    return b + m;
}
```

public void insertApiRoutes(List<ApiRoute> routes) throws SQLException {
    if (routes.isEmpty()) return;
    String sql = """
        INSERT INTO api_routes (symbol_id, http_method, path, base_path, method_path, file_path, start_line)
        VALUES (?, ?, ?, ?, ?, ?, ?)
        """;
    try (PreparedStatement ps = db.getConnection().prepareStatement(sql)) {
        for (ApiRoute r : routes) {
            ps.setLong(1, r.symbolId());
            ps.setString(2, r.httpMethod());
            ps.setString(3, r.path());
            ps.setString(4, r.basePath());
            ps.setString(5, r.methodPath());
            ps.setString(6, r.filePath());
            ps.setInt(7, r.startLine());
            ps.addBatch();
        }
        ps.executeBatch();
        try (ResultSet rs = ps.getGeneratedKeys()) {
            int i = 0;
            while (rs.next() && i < routes.size()) {
                ApiRoute old = routes.get(i);
                routes.set(i, new ApiRoute(rs.getLong(1), old.symbolId(), old.httpMethod(),
                    old.path(), old.basePath(), old.methodPath(), old.filePath(), old.startLine()));
                i++;
            }
        }
    }
}

private List<ApiRoute> mapApiRoutes(PreparedStatement ps) throws SQLException {
    try (ResultSet rs = ps.executeQuery()) {
        List<ApiRoute> routes = new ArrayList<>();
        while (rs.next()) {
            routes.add(new ApiRoute(
                rs.getLong("id"),
                rs.getLong("symbol_id"),
                rs.getString("http_method"),

/**
 * Find all ancestors (super classes and interfaces) of a type.
 * Uses recursive CTE walking super_class chain.

public List<Symbol> findTypeAncestors(String qualifiedName, int maxDepth) throws SQLException {
    String sql = """
        WITH RECURSIVE hierarchy AS (
            SELECT id, file_path, start_line, end_line, kind, name, qualified_name,
                   signature, return_type, parent_class, modifiers, javadoc, super_class, interfaces,
                   is_data_class, is_object, is_sealed, is_companion, is_trait, is_case_class,
                   0 as depth
            FROM symbols
            WHERE qualified_name = ? AND kind = 'CLASS'
            UNION ALL
            SELECT s.id, s.file_path, s.start_line, s.end_line, s.kind, s.name, s.qualified_name,
                   s.signature, s.return_type, s.parent_class, s.modifiers, s.javadoc, s.super_class, s.interfaces,
                   s.is_data_class, s.is_object, s.is_sealed, s.is_companion, s.is_trait, s.is_case_class,
                   h.depth + 1
            FROM symbols s
            INNER JOIN hierarchy h ON s.qualified_name = h.super_class
            WHERE h.depth < ? AND s.kind = 'CLASS'
        )
        SELECT * FROM hierarchy ORDER BY depth
        """;
    try (PreparedStatement ps = db.getConnection().prepareStatement(sql)) {
        ps.setString(1, qualifiedName);
        ps.setInt(2, maxDepth);
        return mapSymbols(ps);
    }
}

/**
 * Find all descendants (sub classes) of a type.
 * Uses iterative Java approach because interfaces field is bracket-comma format.
 */
public List<Symbol> findTypeDescendants(String qualifiedName, int maxDepth) throws SQLException {
    // Start with direct subclasses
    List<Symbol> result = new ArrayList<>();
    List<String> currentLevel = List.of(qualifiedName);

    for (int depth = 0; depth < maxDepth && !currentLevel.isEmpty(); depth++) {
        List<Symbol> nextLevel = new ArrayList<>();
        List<String> nextLevelNames = new ArrayList<>();

        for (String parentName : currentLevel) {
            // Find classes whose super_class = parentName
            String sql = """
                SELECT """ + SYMBOL_COLUMNS + """
                FROM symbols
                WHERE kind = 'CLASS' AND super_class = ?
                """;
            try (PreparedStatement ps = db.getConnection().prepareStatement(sql)) {
                ps.setString(1, parentName);
                List<Symbol> children = mapSymbols(ps);
                for (Symbol child : children) {
                    if (!result.stream().anyMatch(s -> s.id() == child.id())) {
                        result.add(child);
                        nextLevelNames.add(child.qualifiedName());
                    }
                }
            }

            // Find classes whose interfaces list contains parentName
            // interfaces stored as [A, B, C]
            String ifacesSql = """
                SELECT """ + SYMBOL_COLUMNS + """
                FROM symbols
                WHERE kind = 'CLASS'
                  AND (interfaces LIKE '%"' || ? || '"%'
                       OR interfaces = '[' || ? || ']'
                       OR interfaces LIKE '[' || ? || ',%'
                       OR interfaces LIKE '%,' || ? || ']'
                       OR interfaces LIKE '%,' || ? || ',%')
                """;
            try (PreparedStatement ps = db.getConnection().prepareStatement(ifacesSql)) {
                ps.setString(1, parentName);
                ps.setString(2, parentName);
                ps.setString(3, parentName);
                ps.setString(4, parentName);
                ps.setString(5, parentName);
                List<Symbol> children = mapSymbols(ps);
                for (Symbol child : children) {
                    if (!result.stream().anyMatch(s -> s.id() == child.id())) {
                        result.add(child);
                        nextLevelNames.add(child.qualifiedName());
                    }
                }
            }
        }
        currentLevel = nextLevelNames;
    }
    return result;
}

/**
 * Find the full type hierarchy (both ancestors and descendants).
 */
public Map<String, Object> findTypeHierarchy(String qualifiedName, String direction, int maxDepth) throws SQLException {
    List<Symbol> ancestors = List.of();
    List<Symbol> descendants = List.of();

    if ("up".equals(direction) || "both".equals(direction)) {
        ancestors = findTypeAncestors(qualifiedName, maxDepth);
    }
    if ("down".equals(direction) || "both".equals(direction)) {
        descendants = findTypeDescendants(qualifiedName, maxDepth);
    }

    Map<String, Object> result = new LinkedHashMap<>();
    result.put("type", qualifiedName);
    result.put("direction", direction);
    result.put("ancestors", ancestors.stream().map(s -> Map.of(
        "id", s.id(),
        "name", s.name(),
        "qualified_name", s.qualifiedName(),
        "super_class", s.superClass() != null ? s.superClass() : "",
        "interfaces", s.interfaces() != null ? s.interfaces() : List.of(),
        "file", s.filePath(),
        "line", s.startLine()
    )).toList());
    result.put("descendants", descendants.stream().map(s -> Map.of(
        "id", s.id(),
        "name", s.name(),
        "qualified_name", s.qualifiedName(),
        "super_class", s.superClass() != null ? s.superClass() : "",
        "interfaces", s.interfaces() != null ? s.interfaces() : List.of(),
        "file", s.filePath(),
        "line", s.startLine()
    )).toList());
    result.put("ancestor_count", ancestors.size());
    result.put("descendant_count", descendants.size());
    return result;
}

private void extractBeanDependencies(ClassOrInterfaceDeclaration cd,
                                      List<BeanDependency> beanDependencies,
                                      String filePath) {
    String classQn = cd.getFullyQualifiedName().orElse("");
    int classBeginLine = cd.getBegin().map(p -> p.line).orElse(0);

    // Check if this class is a Spring bean
    boolean isSpringBean = hasAnyAnnotation(cd, "Component", "Service", "Repository",
        "Controller", "RestController", "Configuration", "ManagedBean");
    if (!isSpringBean) return;

    // 1. Fields with @Autowired/@Inject
    for (FieldDeclaration fd : cd.getFields()) {
        if (hasAnyAnnotation(fd, "Autowired", "Inject")) {
            String fieldType = fd.getElementType().asString();
            String fieldName = fd.getVariables().stream()
                .findFirst().map(v -> v.getNameAsString()).orElse("");
            int line = fd.getBegin().map(p -> p.line).orElse(0);
            beanDependencies.add(new BeanDependency(0, 0, 0, fieldType, "FIELD", fieldName, filePath, line));
            // Store the owner's qualified name for resolution (use dependsOnType to carry it temporarily)
            // Actually, we need to track which class owns this dependency.
            // Similar to annotations, we need a parallel list or embed the owner info.
        }
    }

    // 2. Constructors with @Autowired/@Inject, or single constructor
    List<ConstructorDeclaration> constructors = cd.getConstructors();
    boolean hasAutowiredConstructor = constructors.stream()
        .anyMatch(c -> hasAnyAnnotation(c, "Autowired", "Inject"));
    boolean singleConstructor = constructors.size() == 1;

    for (ConstructorDeclaration ctor : constructors) {
        if (hasAutowiredConstructor || singleConstructor) {
            if (!hasAnyAnnotation(ctor, "Autowired", "Inject") && !singleConstructor) continue;
            for (Parameter param : ctor.getParameters()) {
                String paramType = param.getType().asString();
                String paramName = param.getNameAsString();
                int line = ctor.getBegin().map(p -> p.line).orElse(0);
                String injType = hasAnyAnnotation(param, "Autowired", "Inject") ? "PARAMETER" : "CONSTRUCTOR";
                beanDependencies.add(new BeanDependency(0, 0, 0, paramType, injType, paramName, filePath, line));
            }
        }
    }

    // 3. Setter methods with @Autowired/@Inject
    for (MethodDeclaration md : cd.getMethods()) {
        if (hasAnyAnnotation(md, "Autowired", "Inject") && md.getParameters().size() == 1) {
            Parameter param = md.getParameters().get(0);
            String paramType = param.getType().asString();
            String paramName = param.getNameAsString();
            int line = md.getBegin().map(p -> p.line).orElse(0);
            beanDependencies.add(new BeanDependency(0, 0, 0, paramType, "SETTER", paramName, filePath, line));
        }
    }
}

private boolean hasAnyAnnotation(NodeWithAnnotations<?> node, String... names) {
    for (String name : names) {
        if (node.getAnnotationByName(name).isPresent()) return true;
    }
    return false;
}

```java
private void extractTestMappings(ClassOrInterfaceDeclaration cd,
                                  List<ImportDeclaration> imports,
                                  List<TestMapping> testMappings,
                                  String filePath) {
    String className = cd.getNameAsString();
    String classQn = cd.getFullyQualifiedName().orElse("");

    // Only process classes that look like test classes
    if (!looksLikeTestClass(className)) return;

    // Derive source class name from naming convention
    String sourceClassName = deriveSourceClassName(className);
    if (sourceClassName == null) return;

    // Check imports for the source class
    boolean foundByImport = imports.stream()
        .anyMatch(imp -> {
            String importName = imp.getNameAsString();
            return importName.endsWith("." + sourceClassName)
                || importName.equals(sourceClassName);
        });

    String mappingType = foundByImport ? "IMPORT" : "NAME_PATTERN";
    testMappings.add(new TestMapping(0, 0, 0, classQn, sourceClassName, mappingType, filePath));
}

private boolean looksLikeTestClass(String className) {
    return className.endsWith("Test") || className.endsWith("Tests")
        || className.endsWith("Spec") || className.startsWith("Test");
}

private String deriveSourceClassName(String testClassName) {
    if (testClassName.endsWith("Tests")) {
        return testClassName.substring(0, testClassName.length() - 5); // "UserTests" → "User"
    }
    if (testClassName.endsWith("Test")) {
        return testClassName.substring(0, testClassName.length() - 4); // "UserServiceTest" → "UserService"
    }
    if (testClassName.endsWith("Spec")) {
        return testClassName.substring(0, testClassName.length() - 4); // "UserServiceSpec" → "UserService"
    }
    if (testClassName.startsWith("Test")) {
        return testClassName.substring(4); // "TestUserService" → "UserService"
    }
    return null;
}
```

private Map<String, Object> callFindRelatedTests(JsonObject args) throws Exception {
    ProjectContext ctx = resolveProject(args);
    StorageService storage = ctx.storage();
    String className = args.get("class_name").getAsString();
    int limit = args.has("limit") ? args.get("limit").getAsInt() : 20;

    // Try exact match first, then try short name
    List<TestMapping> mappings = storage.findTestMappingsBySource(className);
    if (mappings.isEmpty()) {
        // Try by short name (last segment)
        int lastDot = className.lastIndexOf('.');
        if (lastDot > 0) {

        if (sourceId > 0) {

    mappings = mappings.stream().limit(limit).toList();

        "project", resolveProjectName(args),
        "source_class", className,

            map.put("test_class", m.testClassName());
